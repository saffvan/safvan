- conditional statements -
* if statement--

	echo "$num is positive"
  elif [ $num -lt 0 ]; then
	echo "$num is negative"
  else
	echo "$num is zero"
  fi

*switch statement--
variable_name="abc"
case $variable_name in
	"abc")
		echo "is abc"
		;;
	"bbd")
		echo "bbd";;
	"fff")
		echo "fff";;
	*)
		echo "invalid";;
esac
